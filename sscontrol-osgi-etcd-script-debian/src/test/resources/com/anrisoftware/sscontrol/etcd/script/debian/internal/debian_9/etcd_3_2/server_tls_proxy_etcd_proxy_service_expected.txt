[Unit]
Description=Etcd Proxy
After=network.target
After=network-online.target
Wants=network-online.target

[Service]
Type=notify
WorkingDirectory=/var/lib/etcd
EnvironmentFile=-/etc/etcd/etcd-proxy.conf
User=etcd
# set GOMAXPROCS to number of processors
ExecStart=/bin/bash -c "GOMAXPROCS=$(nproc) /usr/local/bin/etcd grpc-proxy start --endpoints=\"${ETCD_PROXY_ENDPOINTS}\" --listen-addr=\"${ETCD_PROXY_LISTEN_ADDR}\" --cert=\"${ETCD_PROXY_CERT}\" --key=\"${ETCD_PROXY_KEY}\" --cacert=\"${ETCD_PROXY_CA}\" "
Restart=on-failure
LimitNOFILE=65536

[Install]
WantedBy=multi-user.target
